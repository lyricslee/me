Problem 1.4¶

Generate a dataset of two-dimensional points, and choose a random line in the plane as your target function f, where one side of the line maps to +1 and the other side to −1. Let the inputs x n ∈ R 2 be random points in the plane, and evaluate the target function f on each x n to get the corresponding output y n = f(x n ).

Experiment with the perceptron learning algorithm in the following settings:

import numpy as np 

def generate_pla_dataset(w, rows_num):
    '''
    TODO:
        generate a linear seprable dataset
    Method:
        random.uniform
    '''
    features_num = len(w) # num of x
    print('features_num=', features_num)
    dataset = np.zeros((rows_num, features_num + 1)) # with +1 row for y(+1 or -1)
    print('dataset shape=', dataset.shape)
    
    for i in range(rows_num):
        x = np.random.uniform(-1, 1, features_num)
        y = np.dot(w, x)
        y = 1 if np.sign(y) > 0 else -1 # sign(y) return +1 or -1
        dataset[i] = np.append(x, y)
    
    return dataset

def pla_train(dataset):
    ''' 
    train pla
    '''
    rows_num = dataset.shape[0]
    features_num = dataset.shape[1]
    w = np.zeros((1, features_num - 1)) # features_num - 1 for deleting y(+1, -1)
    finished = False
    
    i = 0
    count = 0
    while not finished and i < rows_num:
        x = dataset[i, 0:-1]
        y = dataset[i][-1]
        
        if y != np.sign(np.dot(w, x)):
            w = w + y*x
            finished = False
            i = 0;
        else:
            i += 1
        count += 1
    print('run all count=', count)
    return w
